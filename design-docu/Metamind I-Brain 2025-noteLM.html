    def _generate_binary_matrix(self, size):

        """Generates a random binary matrix of the given size."""

        return [[random.randint(0, 1) for _ in range(size)] for _ in range(size)]

    def _rotate_matrix(self, matrix):

        """Rotates a square matrix 90 degrees clockwise."""

        size = len(matrix)

        rotated = [[0 for _ in range(size)] for _ in range(size)]

        for i in range(size):

            for j in range(size):

                rotated[j][size - 1 - i] = matrix[i][j]

        return rotated

        row = random.randint(0, size - 1)

        col = random.randint(0, size - 1)

        matrix[row][col] = 1 - matrix[row][col]  # Flip 0 to 1 or 1 to 0

        return matrix
